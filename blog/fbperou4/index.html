矩阵(Matrix)定义

一个 n×mn\times mn×m 的矩阵是一个由 nnn 行 mmm 列元素排成的矩形阵列。矩阵里的元素可以是数字符号或者数学式。

形如(abcd)\left(\begin{matrix}a&b\\c&d\end{matrix}\right)(ac​bd​)的数表称为二阶矩阵，它由二行二列组成，其中a,b,c,da,b,c,da,b,c,d称为这个矩阵的元素。

形如(x1x2)\left(\begin{matrix}x_1\\x_2\end{matrix}\right)(x1​x2​​)的有序对称为列向量(Column Vector)

先随随便便写一个求快速幂的代码(相信大家都会写吧

int qpow(int a, int k) {
    int res = 1;
    while(k) {
        if(a & 1) res *= a;
        k >>= 1;
        a *= a;
    }
    return res;
}

那么到底怎么用矩阵求斐波那契数列呢？

代码

#include<cstdio>
#include<cstring>

struct Matrix {
	long long fib[2][2];
	Matrix() {memset(fib, 0, sizeof fib);}
};

Matrix qmul(Matrix &a, Matrix &b) {
	Matrix c;
	for(int i = 0; i < 2; ++i) {
		for(int j = 0; j < 2; ++j) {
			for(int k = 0; k < 2; ++k) {
				c.fib[i][j] += a.fib[i][k] * b.fib[k][j];
			}
		}
	}
	return c;
}

Matrix qpow(Matrix a, long long k) {
	Matrix ans;
	ans.fib[0][0] = ans.fib[1][1] = 1;
	while(k) {
		if(k & 1) ans = qmul(ans, a);
		a = qmul(a, a);
		k >>= 1;
	}
	return ans;
}

int main() {
	long long n;
	Matrix a, b;
	scanf("%lld", &n);
	a.fib[0][0] = a.fib[0][1] = a.fib[1][0] = 1;
	b = qpow(a, n);
	printf("%lld\n", b.fib[0][1]);
	return 0;
}
