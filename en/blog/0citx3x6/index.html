[HNOI2008] Toy Packing

Problem Description

Professor P is going to watch the Olympics, but he can't bear to leave his toys behind, so he decides to ship all his toys to Beijing. He uses his own compressor for compression, which can turn any item into a pile and put it into a special one-dimensional container.

Professor P has nnn toys numbered 1⋯n1 \cdots n1⋯n, and the one-dimensional length of the iii-th toy after compression is CiC_iCi​.

For easy organization, Professor P requires:

  • The toy numbers in a one-dimensional container must be consecutive.
  • If there are multiple toys in a one-dimensional container, a unit length of filler must be added between two toys. Formally, if toys from the iii-th to the jjj-th are placed in a container, the container length will be x=j−i+∑k=ijCkx=j-i+\sum_{k=i}^{j}C_kx=j−i+∑k=ij​Ck​.

The cost of making a container is related to the container's length. According to the professor's research, if the container length is xxx, its manufacturing cost is (x−L)2(x-L)^2(x−L)2, where LLL is a constant. Professor P doesn't care about the number of containers; he can make containers of any length, even exceeding LLL. But he hopes to minimize the total cost of all containers.

Input Format

The first line has two integers separated by a space, representing nnn and LLL.

Lines 222 to (n+1)(n+1)(n+1), each line has one integer, where the (i+1)(i+1)(i+1)-th line represents the length CiC_iCi​ of the iii-th toy.

Output Format

Output one line with one integer, representing the minimum total cost of all containers.

il ll sq(ll x) {return x * x;}

int main() {
	scanf("%d%d%d", &n, &L, &C);
	sum[1] = C; f[1] = sq(C - L);
	Qt[++t] = 0; Qx[t] = L + 1; Qy[t] = sq(Qx[t]);
	Qt[++t] = 1; Qx[t] = L + 2 + C; Qy[t] = f[1] + sq(Qx[t]);
	for(int i = 2; i <= n; ++i) {
		scanf("%d", &C);
		sum[i] = sum[i - 1] + C;
		double slope = (i + sum[i]) << 1;
		while(h < t && sl(Qx[h], Qy[h], Qx[h + 1], Qy[h + 1]) < slope) ++h;
		f[i] = f[Qt[h]] + sq(i - Qt[h] - 1 + sum[i] - sum[Qt[h]] - L);

The Problem Arises

What if CkC_kCk​ can be less than 0? How should we handle this?

We can immediately see that this does not satisfy monotonicity.

What if during an exam you can't prove whether the slope points or decisions have/don't have monotonicity?

There are two methods. The first is to generate test cases and try several sets.

The second is to just use the approach for when monotonicity is not satisfied, regardless. After all, the approach for when monotonicity is not satisfied can also be used when monotonicity is satisfied.

Approaches when monotonicity is not satisfied:

Divide and Conquer

We use dc(l,r)dc(l, r)dc(l,r) to represent computing dpidp_idpi​ in [l,r][l, r][l,r].

For the left half, we first use dc(l,mid)dc(l, mid)dc(l,mid) to compute dpidp_idpi​ in [l,mid][l, mid][l,mid]. Then we know all the decision points, so we can build a convex hull. Then we use this convex hull to update dpidp_idpi​ in [mid,r][mid, r][mid,r]. At this point the convex hull is fixed, so we can sort the query slopes for [mid,r][mid, r][mid,r] and maintain them with a monotonic queue. Of course, we can also directly binary search on the convex hull.

For each dpidp_idpi​ in [mid,r][mid, r][mid,r], if its optimal decision point is in [1,mid][1, mid][1,mid], it has already been updated in the previous step. If its optimal decision point is not in [l,mid][l, mid][l,mid], we don't need the convex hull from [l,mid][l, mid][l,mid]. So we can directly discard the left convex hull and use dc(mid+1,r)dc(mid + 1, r)dc(mid+1,r) to compute the subsequent dpidp_idpi​.

#include <vector>
#include <stdio.h>
#include <string.h>
#include <algorithm>

int n, h[100005], w[100005], pre[100005];
long long dp[100005], sum[100005];

struct Point {
	long long x, y;
	int id;
	Point(long long _x, long long _y, int _id): x(_x), y(_y), id(_id) {}
	double operator/ (const Point &p) const { return (double) (y - p.y) / (x - p.x); }
};

long long min(long long a, long long b) { return a < b ? a : b; }

void cdq(int l, int r) {
	while (l != r) {
		std::vector<Point> Q, temp;
		int mid = (l + r) >> 1;
		cdq(l, mid);
		for (int i = l; i <= mid; ++i)
			temp.push_back(Point(h[i], dp[i] + (long long) h[i] * h[i] - sum[i], i));
		std::sort(temp.begin(), temp.end(), [] (const Point &p1, const Point &p2) {
			return p1.x == p2.x ? p1.y < p2.y : p1.x < p2.x;
		});
		Q.push_back(temp[0]); int _size = 0;
		for (int i = 1; i < (int) temp.size(); ++i) {
			if (Q[_size].x == temp[i].x) continue;
			while (_size && Q[_size] / Q[_size - 1] > Q[_size] / temp[i]) Q.pop_back(), --_size;
			Q.push_back(temp[i]); ++_size;
		}
		for (int i = r; i > mid; --i) {
			int l = 1, r = Q.size() - 1, sl = h[i] << 1;
			while (l <= r) {
				int mid = (l + r) >> 1;
				if (Q[mid] / Q[mid - 1] > sl) r = mid - 1;
				else l = mid + 1;
			}
			int k = Q[r].id;
			dp[i] = min(dp[i], dp[k] + (long long) (h[i] - h[k]) * (h[i] - h[k]) + sum[i - 1] - sum[k]);
		}
		l = mid + 1;
	}
}

	memset(dp, 0x3f, sizeof dp); dp[1] = 0;
	cdq(1, n); printf("%lld\n", dp[n]);
	return 0;
}

Balanced Tree

The idea is concise, easy to understand once explained, but difficult to implement correctly.

Code

#include <stdio.h>
#include <string.h>

inline unsigned rand() {
	static unsigned seed = 19260817;
	return (seed = seed * 1279u + 10001279u) ^= (seed >> 7);
}

struct Point {
	long long x, y;
	int id;
	Point(long long _x, long long _y, int _id): x(_x), y(_y), id(_id) {}
	Point() {}
	double operator/ (const Point &p) const { return (double) (y - p.y) / (x - p.x); }
};

struct Treap {
	struct node {
		int ls, rs, size;
		Point p;
		double lp, rp;
		unsigned key;
	} tr[100005];
	int cnt, root;
	inline int new_node(const Point &p) {
		tr[++cnt].p = p;
		tr[cnt].key = rand();
		tr[cnt].size = 1;
		return cnt;
	}
	inline void push_up(int k) { if (k) tr[k].size = tr[tr[k].ls].size + tr[tr[k].rs].size + 1; }
	void rotate_left(int &k) {
		int old_k = k;
		k = tr[k].rs;
		tr[old_k].rs = tr[k].ls;
		tr[k].ls = old_k;
		push_up(old_k);
		push_up(k);
	}
	void rotate_right(int &k) {
		int old_k = k;
		k = tr[k].ls;
		tr[old_k].ls = tr[k].rs;
		tr[k].rs = old_k;
		push_up(old_k);
		push_up(k);
	}
	int insert(int &k, const Point &v) {
		int tmp;
		if (!k) { return k = new_node(v); }
		if (v.x < tr[k].p.x) {
			if (tmp = insert(tr[k].ls, v)) {
				if (tr[k].key > tr[tr[k].ls].key) rotate_right(k);
				push_up(k);
				return tmp;
			} else return 0;
		} else if (v.x > tr[k].p.x) {
			if (tmp = insert(tr[k].rs, v)) {
				if (tr[k].key > tr[tr[k].rs].key) rotate_left(k);
				push_up(k);
				return tmp;
			} else return 0;
		} else {
			if (tr[k].p.y < v.y) return 0;
			tr[k].p.y = v.y; tr[k].p.id = v.id; return k;
		}
	}
	void remove(int &k, long long x) {
		if (!k) return;
		if (tr[k].p.x == x) {
			if (tr[k].ls && tr[k].rs) {
				if (tr[tr[k].ls].key < tr[tr[k].rs].key) rotate_right(k), remove(tr[k].rs, x);
				else rotate_left(k), remove(tr[k].ls, x);
			} else k = tr[k].ls | tr[k].rs;
		} else if (x < tr[k].p.x) remove(tr[k].ls, x);
		else remove(tr[k].rs, x);
		push_up(k);
	}
	int rank(long long x) {
		int rnk = 0, k = root;
		while (k) {
			if (x <= tr[k].p.x) k = tr[k].ls;
			else rnk += tr[tr[k].ls].size + 1, k = tr[k].rs;
		}
		return rnk + 1;
	}
	int value(int x) {
		if (x > tr[root].size || x <= 0) return -1;
		int k = root;
		while (k) {
			if (x <= tr[tr[k].ls].size) k = tr[k].ls;
			else if (x == tr[tr[k].ls].size + 1) return k;
			else x -= tr[tr[k].ls].size + 1, k = tr[k].rs;
		}
		return -1;
	}
	int prev(long long v) { return value(rank(v) - 1); }
	int next(long long v) { return value(rank(v + 1)); }
	void insert_hull(const Point &point) {
		int k = insert(root, point);
		if (!k) return;
		int k1 = prev(point.x);
		if (k1 != -1) {
			while (tr[k1].lp > point / tr[k1].p) remove(root, tr[k1].p.x), k1 = prev(tr[k1].p.x);
			tr[k].lp = tr[k1].rp = point / tr[k1].p;
		} else tr[k].lp = -1e30;
		int k2 = next(point.x);
		if (k2 != -1) {
			while (point / tr[k2].p > tr[k2].rp)
				remove(root, tr[k2].p.x), k2 = next(tr[k2].p.x);
			tr[k].rp = tr[k2].lp = tr[k2].p / point;
		} else tr[k].rp = 1e30;
		if (tr[k].lp > tr[k].rp) {
			remove(root, tr[k].p.x);
			tr[k1].rp = tr[k2].lp = tr[k1].p / tr[k2].p;
		}
	}
	int query_hull(long long slope) {
		int k = root;
		while (k) {
			if (tr[k].lp <= slope && slope <= tr[k].rp) return tr[k].p.id;
			else if (slope < tr[k].lp) k = tr[k].ls;
			else k = tr[k].rs;
		}
		return -1;
	}
} treap;

int n, w[100005], h[100005];
long long sum[100005], dp[100005];

long long min(long long a, long long b) { return a < b ? a : b; }

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) scanf("%d", h + i);
	for (int i = 1; i <= n; ++i) scanf("%d", w + i), sum[i] = sum[i - 1] + w[i];
	memset(dp, 0x3f, sizeof dp); dp[1] = 0;
	treap.new_node(Point(h[1], (long long) h[1] * h[1] - w[1], 1));
	treap.root = 1; treap.tr[1].lp = -1e30; treap.tr[1].rp = 1e30;
	for (int i = 2; i <= n; ++i) {
		int j = treap.query_hull(h[i] << 1);
		dp[i] = min(dp[i], dp[j] + (long long) (h[i] - h[j]) * (h[i] - h[j]) + sum[i - 1] - sum[j]);
		treap.insert_hull(Point(h[i], dp[i] + (long long) h[i] * h[i] - sum[i], i));
	}
	printf("%lld", dp[n]);
	return 0;
}

Binary Grouping

We divide all points into several groups, where the iii-th group has size 2i2^i2i. Obviously, the number of groups does not exceed O(log⁡n)O(\log n)O(logn). We maintain the convex hull for each group.

When inserting a point, if there is no group 0, make it group 0. Otherwise, merge it with group 0. This way we get group 1. Next, if there is a group 1, merge it with group 1 to get group 2... similar to the process of adding 1 in binary.

For queries, binary search in each group separately and take the maximum.

Regarding time complexity:

Obviously, each point is merged at most O(log⁡n)O(\log n)O(logn) times. The time complexity of merging two convex hulls containing O(n)O(n)O(n) nodes is O(n)O(n)O(n), so amortized to each point, the merge time complexity is O(1)O(1)O(1). Since there are O(n)O(n)O(n) points in total, the total time complexity is O(nlog⁡n)O(n\log n)O(nlogn).

Code

#include <string.h>
#include <stdio.h>
#include <vector>

int n, h[100005], w[100005];
long long dp[100005], sum[100005];

struct Point {
	long long x, y;
	int id;
	Point(long long _x, long long _y, int _id): x(_x), y(_y), id(_id) {}
	double operator/ (const Point &p) const { return (double) (y - p.y) / (x - p.x); }
};

struct ConvexHull {
	std::vector<Point> p;
	void merge(ConvexHull &h) {
		std::vector<Point> possible_conv;
		int i = 0, j = 0;
		while (i < (int) p.size() && j < (int) h.p.size())
			if (p[i].x == h.p[j].x) {
				if (p[i].y < h.p[j].y) possible_conv.push_back(p[i]);
				else possible_conv.push_back(h.p[j]);
				++i; ++j;
			} else if (p[i].x < h.p[j].x) possible_conv.push_back(p[i]), ++i;
			else possible_conv.push_back(h.p[j]), ++j;
		while (i < (int) p.size()) possible_conv.push_back(p[i]), ++i;
		while (j < (int) h.p.size()) possible_conv.push_back(h.p[j]), ++j;
		p.clear(); p.push_back(possible_conv[0]);
		int _size = 0;
		for (int i = 1; i < (int) possible_conv.size(); ++i) {
			while (_size && p[_size] / p[_size - 1] > p[_size] / possible_conv[i]) p.pop_back(), --_size;
			p.push_back(possible_conv[i]); ++_size;
		}
		h.p.clear();
	}
	int query(long long k) {
		int l = 1, r = p.size() - 1;
		while (l <= r) {
			int mid = (l + r) >> 1;
			if (p[mid] / p[mid - 1] > k) r = mid - 1;
			else l = mid + 1;
		}
		return p[r].id;
	}
	ConvexHull() {}
	ConvexHull(const Point &P) { p.push_back(P); }
} conv[17];

long long min(long long a, long long b) { return a < b ? a : b; }

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) scanf("%d", h + i);
	for (int i = 1; i <= n; ++i) scanf("%d", w + i), sum[i] = sum[i - 1] + w[i];
	memset(dp, 0x3f, sizeof dp); dp[1] = 0;
	conv[0] = ConvexHull(Point(h[1], (long long) h[1] * h[1] - w[1], 1));
	for (int i = 2; i <= n; ++i) {
		for (int j = 0; j < 17; ++j) {
			if (conv[j].p.empty()) continue;
			int k = conv[j].query(h[i] << 1);
			dp[i] = min(dp[i], dp[k] + (long long) (h[i] - h[k]) * (h[i] - h[k]) + sum[i - 1] - sum[k]);
		}
		ConvexHull cv(Point(h[i], dp[i] + (long long) h[i] * h[i] - sum[i], i));
		for (int j = 0; j < 17; ++j) {
			if (conv[j].p.empty()) {
				conv[j].p = cv.p;
				break;
			}
			cv.merge(conv[j]);
		}
	}
	printf("%lld", dp[n]);
	return 0;
}

Summary

In DP minimization/maximization problems, we can divide the transition equation into four parts: one part independent of both iii and jjj, one part dependent only on jjj, one part dependent only on iii, and one part dependent on both iii and jjj. If it can be written in the form y=kx+by=kx+by=kx+b, we can maintain a convex hull to perform state transitions.
