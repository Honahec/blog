Matrix Definition

An n×mn\times mn×m matrix is a rectangular array of elements arranged in nnn rows and mmm columns. The elements in a matrix can be numbers, symbols, or mathematical expressions.

A table in the form of (abcd)\left(\begin{matrix}a&b\\c&d\end{matrix}\right)(ac​bd​) is called a second-order matrix, which consists of two rows and two columns, where a,b,c,da,b,c,da,b,c,d are called the elements of this matrix.

An ordered pair in the form of (x1x2)\left(\begin{matrix}x_1\\x_2\end{matrix}\right)(x1​x2​​) is called a column vector.

int qpow(int a, int k) {
    int res = 1;
    while(k) {
        if(a & 1) res *= a;
        k >>= 1;
        a *= a;
    }
    return res;
}

So how exactly do we use matrices to calculate the Fibonacci sequence?

Code

#include<cstdio>
#include<cstring>

struct Matrix {
	long long fib[2][2];
	Matrix() {memset(fib, 0, sizeof fib);}
};

Matrix qmul(Matrix &a, Matrix &b) {
	Matrix c;
	for(int i = 0; i < 2; ++i) {
		for(int j = 0; j < 2; ++j) {
			for(int k = 0; k < 2; ++k) {
				c.fib[i][j] += a.fib[i][k] * b.fib[k][j];
			}
		}
	}
	return c;
}

Matrix qpow(Matrix a, long long k) {
	Matrix ans;
	ans.fib[0][0] = ans.fib[1][1] = 1;
	while(k) {
		if(k & 1) ans = qmul(ans, a);
		a = qmul(a, a);
		k >>= 1;
	}
	return ans;
}

int main() {
	long long n;
	Matrix a, b;
	scanf("%lld", &n);
	a.fib[0][0] = a.fib[0][1] = a.fib[1][0] = 1;
	b = qpow(a, n);
	printf("%lld\n", b.fib[0][1]);
	return 0;
}
